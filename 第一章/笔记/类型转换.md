``````python
#!/usr/bin/env python
# coding: utf-8

"""类型转换

    # round函数 #
    round函数可以用来将浮点数转换为最近的整数


    # math.floor函数  #
    floor函数用来将浮点数转换为整数，舍弃小数部分
    函数在math模块内，使用时需要使用import 导入math模块


    # math.ceil函数 #
    与floor函数相反用于将整数与小数部分加起来（把小数当作1）

"""

import math

f1 = 1.2
f2 = 2.8

round(f1)           # => 1
math.floor(f1)      # => 1
math.ceil(f1)       # => 2

round(f2)           # => 3
math.floor(f2)      # => 2
math.ceil(f2)       # => 3
``````
